# ===== One-time setup (only if you haven't created the venv before) =====
cd C:\Users\bharg\Downloads\final_agentic_ai
py -m venv .venv
.\.venv\Scripts\pip.exe install -r requirements.txt

# ===== Start Backend API (optional; website can run without backend) =====
# In a terminal window:
cd C:\Users\bharg\Downloads\final_agentic_ai
.\.venv\Scripts\python.exe -m uvicorn app.main:app --host 0.0.0.0 --port 8000
# Browse: http://localhost:8000  (Docs: http://localhost:8000/docs)

# ===== Start Static Website =====
# In a separate terminal window:
cd C:\Users\bharg\Downloads\final_agentic_ai\web
py -m http.server 8080
# Browse: http://localhost:8080

# Notes:
# - The website computes allocations locally (no API required). Running the API is optional.
# - If 'py' is not recognized, replace 'py' with 'python' in the commands above.
